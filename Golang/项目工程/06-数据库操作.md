# 数据库操作

> **目标**：掌握 Go 的 database/sql 标准库和 GORM ORM 框架，能完成 MySQL/PostgreSQL 的 CRUD、事务、连接池配置，以及 Redis 操作。

## 一、database/sql 标准库

### 1.1 连接数据库

```go
import (
    "database/sql"
    _ "github.com/go-sql-driver/mysql" // MySQL 驱动（匿名导入，注册驱动）
)

func initDB() (*sql.DB, error) {
    // DSN 格式: user:password@tcp(host:port)/dbname?params
    dsn := "root:password@tcp(127.0.0.1:3306)/mydb?charset=utf8mb4&parseTime=true"
    db, err := sql.Open("mysql", dsn)
    if err != nil {
        return nil, err
    }

    // 配置连接池
    db.SetMaxOpenConns(25)                 // 最大打开连接数
    db.SetMaxIdleConns(10)                 // 最大空闲连接数
    db.SetConnMaxLifetime(5 * time.Minute) // 连接最大存活时间

    // 测试连接
    if err := db.Ping(); err != nil {
        return nil, err
    }
    return db, nil
}
```

**关键点**：
- `sql.Open` 不会立即建立连接，只是初始化连接池
- `sql.DB` 是并发安全的，整个应用共享一个实例
- 驱动通过 `init()` 函数自注册，用 `_` 匿名导入

### 1.2 查询

```go
type User struct {
    ID    int
    Name  string
    Email string
}

// 查询单行
func getUser(db *sql.DB, id int) (*User, error) {
    var u User
    err := db.QueryRow("SELECT id, name, email FROM users WHERE id = ?", id).
        Scan(&u.ID, &u.Name, &u.Email)
    if err == sql.ErrNoRows {
        return nil, nil // 未找到
    }
    return &u, err
}

// 查询多行
func listUsers(db *sql.DB) ([]User, error) {
    rows, err := db.Query("SELECT id, name, email FROM users ORDER BY id")
    if err != nil {
        return nil, err
    }
    defer rows.Close() // 必须关闭！否则连接泄漏

    var users []User
    for rows.Next() {
        var u User
        if err := rows.Scan(&u.ID, &u.Name, &u.Email); err != nil {
            return nil, err
        }
        users = append(users, u)
    }
    return users, rows.Err() // 检查遍历过程中的错误
}
```

### 1.3 增删改

```go
// 插入
func createUser(db *sql.DB, name, email string) (int64, error) {
    result, err := db.Exec("INSERT INTO users (name, email) VALUES (?, ?)", name, email)
    if err != nil {
        return 0, err
    }
    return result.LastInsertId()
}

// 更新
func updateUser(db *sql.DB, id int, name string) error {
    result, err := db.Exec("UPDATE users SET name = ? WHERE id = ?", name, id)
    if err != nil {
        return err
    }
    rows, _ := result.RowsAffected()
    if rows == 0 {
        return fmt.Errorf("user %d not found", id)
    }
    return nil
}

// 删除
func deleteUser(db *sql.DB, id int) error {
    _, err := db.Exec("DELETE FROM users WHERE id = ?", id)
    return err
}
```

### 1.4 预编译语句

```go
stmt, err := db.Prepare("SELECT id, name FROM users WHERE email = ?")
if err != nil {
    return err
}
defer stmt.Close()

// 可以多次执行
var u User
err = stmt.QueryRow("alice@example.com").Scan(&u.ID, &u.Name)
```

### 1.5 事务

```go
func transfer(db *sql.DB, fromID, toID int, amount float64) error {
    tx, err := db.Begin()
    if err != nil {
        return err
    }
    // defer 处理：成功 commit，失败 rollback
    defer func() {
        if err != nil {
            tx.Rollback()
        }
    }()

    // 扣款
    _, err = tx.Exec("UPDATE accounts SET balance = balance - ? WHERE id = ?", amount, fromID)
    if err != nil {
        return err
    }

    // 加款
    _, err = tx.Exec("UPDATE accounts SET balance = balance + ? WHERE id = ?", amount, toID)
    if err != nil {
        return err
    }

    return tx.Commit()
}
```

## 二、GORM

### 2.1 安装

```bash
go get -u gorm.io/gorm
go get -u gorm.io/driver/mysql   # MySQL
go get -u gorm.io/driver/postgres # PostgreSQL
```

### 2.2 连接

```go
import (
    "gorm.io/driver/mysql"
    "gorm.io/gorm"
    "gorm.io/gorm/logger"
)

func initGorm() (*gorm.DB, error) {
    dsn := "root:password@tcp(127.0.0.1:3306)/mydb?charset=utf8mb4&parseTime=true"
    db, err := gorm.Open(mysql.Open(dsn), &gorm.Config{
        Logger: logger.Default.LogMode(logger.Info), // 打印 SQL
    })
    if err != nil {
        return nil, err
    }

    // 获取底层 sql.DB 配置连接池
    sqlDB, _ := db.DB()
    sqlDB.SetMaxOpenConns(25)
    sqlDB.SetMaxIdleConns(10)
    sqlDB.SetConnMaxLifetime(5 * time.Minute)

    return db, nil
}
```

### 2.3 模型定义

```go
type User struct {
    ID        uint           `gorm:"primaryKey"`
    Name      string         `gorm:"size:100;not null"`
    Email     string         `gorm:"uniqueIndex;size:200"`
    Age       int            `gorm:"default:0"`
    Role      string         `gorm:"type:enum('admin','user');default:'user'"`
    Profile   Profile        `gorm:"foreignKey:UserID"` // 一对一
    Posts     []Post         `gorm:"foreignKey:AuthorID"` // 一对多
    CreatedAt time.Time
    UpdatedAt time.Time
    DeletedAt gorm.DeletedAt `gorm:"index"` // 软删除
}

type Profile struct {
    ID     uint
    UserID uint
    Bio    string
}

type Post struct {
    ID       uint
    Title    string
    AuthorID uint
}

// 自动迁移（建表/修改表结构）
db.AutoMigrate(&User{}, &Profile{}, &Post{})
```

### 2.4 GORM CRUD

```go
// 创建
user := User{Name: "Alice", Email: "alice@example.com", Age: 25}
result := db.Create(&user)
// user.ID 已自动填充
// result.Error, result.RowsAffected

// 批量创建
users := []User{{Name: "Bob"}, {Name: "Charlie"}}
db.Create(&users)

// 查询
var user User
db.First(&user, 1)                                  // 主键查询
db.First(&user, "email = ?", "alice@example.com")   // 条件查询
db.Where("age > ? AND role = ?", 18, "user").Find(&users) // 多条件

// 高级查询
var users []User
db.Select("name, email").
    Where("age > ?", 18).
    Order("created_at DESC").
    Limit(10).Offset(20).
    Find(&users)

// 更新
db.Model(&user).Update("name", "Alice Updated")
db.Model(&user).Updates(User{Name: "New Name", Age: 30}) // 零值不更新
db.Model(&user).Updates(map[string]any{"name": "New", "age": 0}) // map 可更新零值

// 删除
db.Delete(&user, 1) // 软删除（如果有 DeletedAt 字段）
db.Unscoped().Delete(&user, 1) // 真删除
```

### 2.5 关联查询

```go
// 预加载
var user User
db.Preload("Profile").Preload("Posts").First(&user, 1)

// 嵌套预加载
db.Preload("Posts.Comments").First(&user, 1)

// 条件预加载
db.Preload("Posts", "published = ?", true).First(&user, 1)

// Joins 预加载（一条 SQL）
db.Joins("Profile").First(&user, 1)
```

### 2.6 事务

```go
// 方式 1：自动事务
err := db.Transaction(func(tx *gorm.DB) error {
    if err := tx.Create(&user1).Error; err != nil {
        return err // 返回 error 自动回滚
    }
    if err := tx.Create(&user2).Error; err != nil {
        return err
    }
    return nil // 返回 nil 自动提交
})

// 方式 2：手动事务
tx := db.Begin()
if err := tx.Create(&user1).Error; err != nil {
    tx.Rollback()
    return err
}
tx.Commit()
```

### 2.7 Scopes（可复用查询）

```go
func Paginate(page, size int) func(db *gorm.DB) *gorm.DB {
    return func(db *gorm.DB) *gorm.DB {
        offset := (page - 1) * size
        return db.Offset(offset).Limit(size)
    }
}

func Active(db *gorm.DB) *gorm.DB {
    return db.Where("status = ?", "active")
}

// 使用
db.Scopes(Active, Paginate(2, 10)).Find(&users)
```

## 三、Redis（go-redis）

### 3.1 安装

```bash
go get github.com/redis/go-redis/v9
```

### 3.2 连接

```go
import "github.com/redis/go-redis/v9"

func initRedis() *redis.Client {
    rdb := redis.NewClient(&redis.Options{
        Addr:     "localhost:6379",
        Password: "",
        DB:       0,
        PoolSize: 10,
    })

    ctx := context.Background()
    if err := rdb.Ping(ctx).Err(); err != nil {
        log.Fatal("redis connect failed:", err)
    }
    return rdb
}
```

### 3.3 基本操作

```go
ctx := context.Background()

// String
rdb.Set(ctx, "key", "value", 10*time.Minute) // 设置，带过期
val, err := rdb.Get(ctx, "key").Result()
if err == redis.Nil {
    // key 不存在
}

// Hash
rdb.HSet(ctx, "user:1", "name", "Alice", "age", 25)
name, _ := rdb.HGet(ctx, "user:1", "name").Result()
all, _ := rdb.HGetAll(ctx, "user:1").Result() // map[string]string

// List
rdb.RPush(ctx, "queue", "task1", "task2")
val, _ = rdb.LPop(ctx, "queue").Result()

// Set
rdb.SAdd(ctx, "tags", "go", "redis")
members, _ := rdb.SMembers(ctx, "tags").Result()

// Sorted Set
rdb.ZAdd(ctx, "leaderboard", redis.Z{Score: 100, Member: "Alice"})
rdb.ZAdd(ctx, "leaderboard", redis.Z{Score: 200, Member: "Bob"})
top, _ := rdb.ZRevRangeWithScores(ctx, "leaderboard", 0, 9).Result()

// 过期
rdb.Expire(ctx, "key", 1*time.Hour)
rdb.TTL(ctx, "key")
```

### 3.4 缓存模式

```go
func GetUser(ctx context.Context, rdb *redis.Client, db *gorm.DB, id int) (*User, error) {
    cacheKey := fmt.Sprintf("user:%d", id)

    // 1. 查缓存
    data, err := rdb.Get(ctx, cacheKey).Bytes()
    if err == nil {
        var user User
        json.Unmarshal(data, &user)
        return &user, nil
    }

    // 2. 缓存未命中，查数据库
    var user User
    if err := db.First(&user, id).Error; err != nil {
        return nil, err
    }

    // 3. 写入缓存
    encoded, _ := json.Marshal(user)
    rdb.Set(ctx, cacheKey, encoded, 30*time.Minute)

    return &user, nil
}
```

### 3.5 分布式锁

```go
func acquireLock(ctx context.Context, rdb *redis.Client, key string, ttl time.Duration) (bool, error) {
    return rdb.SetNX(ctx, "lock:"+key, "1", ttl).Result()
}

func releaseLock(ctx context.Context, rdb *redis.Client, key string) {
    rdb.Del(ctx, "lock:"+key)
}
```

## 四、数据库迁移

### 4.1 golang-migrate

```bash
go install -tags 'mysql' github.com/golang-migrate/migrate/v4/cmd/migrate@latest

# 创建迁移文件
migrate create -ext sql -dir migrations -seq create_users_table
```

```sql
-- migrations/000001_create_users_table.up.sql
CREATE TABLE users (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(200) UNIQUE NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- migrations/000001_create_users_table.down.sql
DROP TABLE IF EXISTS users;
```

```bash
# 执行迁移
migrate -database "mysql://root:password@tcp(localhost:3306)/mydb" -path migrations up
migrate -database "..." -path migrations down 1  # 回滚 1 步
```

## 速查表

| 操作 | database/sql | GORM |
|------|-------------|------|
| 连接 | `sql.Open(driver, dsn)` | `gorm.Open(driver.Open(dsn))` |
| 查单行 | `QueryRow().Scan()` | `db.First(&u, id)` |
| 查多行 | `Query()` + `rows.Next()` | `db.Find(&users)` |
| 插入 | `Exec("INSERT...")` | `db.Create(&u)` |
| 更新 | `Exec("UPDATE...")` | `db.Model(&u).Updates(...)` |
| 删除 | `Exec("DELETE...")` | `db.Delete(&u, id)` |
| 事务 | `db.Begin()` + `Commit/Rollback` | `db.Transaction(func(tx)...)` |
| 连接池 | `SetMaxOpenConns()` 等 | 通过 `db.DB()` 获取 |

| Redis 类型 | 命令 |
|-----------|------|
| String | `Set/Get/SetNX/Incr` |
| Hash | `HSet/HGet/HGetAll` |
| List | `RPush/LPop/LRange` |
| Set | `SAdd/SMembers/SIsMember` |
| ZSet | `ZAdd/ZRangeWithScores/ZRevRange` |
