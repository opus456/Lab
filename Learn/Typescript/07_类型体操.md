# 类型体操（Type-Level Programming）

类型体操是 TypeScript 高级类型编程的俗称，指利用 TypeScript 类型系统的图灵完备特性，在类型层面进行复杂的逻辑运算。掌握类型体操能让你写出更精确、更安全的类型定义。

---

## 1. 内置工具类型全解

TypeScript 内置了大量工具类型，理解它们的实现原理是掌握类型体操的基础。

### 1.1 Partial\<T\>

将类型 T 的所有属性变为可选。

```typescript
// 实现原理
type MyPartial<T> = {
  [P in keyof T]?: T[P];
};

// 使用场景：更新对象时只需传入部分字段
interface User {
  name: string;
  age: number;
  email: string;
}

function updateUser(id: number, updates: Partial<User>) {
  // updates 中的每个字段都是可选的
}

updateUser(1, { name: "Alice" }); // 只更新 name
```

### 1.2 Required\<T\>

将类型 T 的所有属性变为必选，与 Partial 相反。

```typescript
// 实现原理：-? 表示移除可选修饰符
type MyRequired<T> = {
  [P in keyof T]-?: T[P];
};

// 使用场景：确保配置对象所有字段都已填充
interface Config {
  host?: string;
  port?: number;
  debug?: boolean;
}

// 经过默认值填充后，所有字段都确定存在
function createServer(config: Required<Config>) {
  console.log(config.host); // string，不是 string | undefined
}
```

### 1.3 Readonly\<T\>

将类型 T 的所有属性变为只读。

```typescript
// 实现原理
type MyReadonly<T> = {
  readonly [P in keyof T]: T[P];
};

// 使用场景：防止意外修改状态
interface State {
  count: number;
  items: string[];
}

function getState(): Readonly<State> {
  return { count: 0, items: [] };
}

const state = getState();
// state.count = 1; // Error: Cannot assign to 'count' because it is a read-only property
```

### 1.4 Pick\<T, K\>

从类型 T 中选取指定属性 K 组成新类型。

```typescript
// 实现原理
type MyPick<T, K extends keyof T> = {
  [P in K]: T[P];
};

// 使用场景：只暴露部分字段
interface Article {
  id: number;
  title: string;
  content: string;
  author: string;
  createdAt: Date;
}

// 列表页只需要摘要信息
type ArticleSummary = Pick<Article, "id" | "title" | "author">;
```

### 1.5 Omit\<T, K\>

从类型 T 中排除指定属性 K。

```typescript
// 实现原理：组合 Pick 和 Exclude
type MyOmit<T, K extends keyof any> = Pick<T, Exclude<keyof T, K>>;

// 使用场景：创建时不需要 id 字段
type CreateArticleDTO = Omit<Article, "id" | "createdAt">;
```

### 1.6 Record\<K, T\>

构造一个属性键为 K、属性值为 T 的类型。

```typescript
// 实现原理
type MyRecord<K extends keyof any, T> = {
  [P in K]: T;
};

// 使用场景：定义字典/映射类型
type PageInfo = { title: string; url: string };
type Pages = "home" | "about" | "contact";

const nav: Record<Pages, PageInfo> = {
  home: { title: "首页", url: "/" },
  about: { title: "关于", url: "/about" },
  contact: { title: "联系", url: "/contact" },
};
```

### 1.7 Exclude\<T, U\>

从联合类型 T 中排除可以赋值给 U 的类型。

```typescript
// 实现原理：利用条件类型的分布式特性
type MyExclude<T, U> = T extends U ? never : T;

// 分布式条件类型会逐个处理联合类型的成员
// Exclude<"a" | "b" | "c", "a"> 
// = ("a" extends "a" ? never : "a") | ("b" extends "a" ? never : "b") | ("c" extends "a" ? never : "c")
// = never | "b" | "c"
// = "b" | "c"

type T1 = Exclude<"a" | "b" | "c", "a" | "b">; // "c"
type T2 = Exclude<string | number | boolean, string>; // number | boolean
```

### 1.8 Extract\<T, U\>

从联合类型 T 中提取可以赋值给 U 的类型。

```typescript
// 实现原理
type MyExtract<T, U> = T extends U ? T : never;

// 使用场景：提取特定类型
type T3 = Extract<"a" | "b" | "c", "a" | "f">; // "a"
type T4 = Extract<string | number | (() => void), Function>; // () => void
```

### 1.9 NonNullable\<T\>

从类型 T 中排除 null 和 undefined。

```typescript
// 实现原理
type MyNonNullable<T> = T extends null | undefined ? never : T;

// 使用场景：确保值不为空
type T5 = NonNullable<string | number | null | undefined>; // string | number

function processValue(value: NonNullable<string | null>) {
  // value 一定是 string
  console.log(value.toUpperCase());
}
```

### 1.10 ReturnType\<T\>

获取函数类型 T 的返回值类型。

```typescript
// 实现原理：使用 infer 推断返回值类型
type MyReturnType<T extends (...args: any) => any> = 
  T extends (...args: any) => infer R ? R : any;

// 使用场景：获取函数返回类型而不需要手动声明
function createUser() {
  return { id: 1, name: "Alice", role: "admin" as const };
}

type UserResult = ReturnType<typeof createUser>;
// { id: number; name: string; role: "admin" }
```

### 1.11 Parameters\<T\>

获取函数类型 T 的参数类型，返回元组。

```typescript
// 实现原理
type MyParameters<T extends (...args: any) => any> = 
  T extends (...args: infer P) => any ? P : never;

// 使用场景：包装函数时保持参数类型一致
function request(url: string, method: "GET" | "POST", body?: object) {
  // ...
}

type RequestParams = Parameters<typeof request>;
// [url: string, method: "GET" | "POST", body?: object]

// 用于创建包装函数
function loggedRequest(...args: Parameters<typeof request>) {
  console.log("Requesting:", args[0]);
  return request(...args);
}
```

### 1.12 ConstructorParameters\<T\>

获取构造函数的参数类型。

```typescript
// 实现原理
type MyConstructorParameters<T extends abstract new (...args: any) => any> = 
  T extends abstract new (...args: infer P) => any ? P : never;

// 使用场景
class Animal {
  constructor(public name: string, public age: number) {}
}

type AnimalParams = ConstructorParameters<typeof Animal>; // [name: string, age: number]
```

### 1.13 InstanceType\<T\>

获取构造函数的实例类型。

```typescript
// 实现原理
type MyInstanceType<T extends abstract new (...args: any) => any> = 
  T extends abstract new (...args: any) => infer R ? R : any;

// 使用场景：工厂模式中动态创建实例
function createInstance<T extends new (...args: any[]) => any>(
  Ctor: T,
  ...args: ConstructorParameters<T>
): InstanceType<T> {
  return new Ctor(...args);
}

const animal = createInstance(Animal, "Dog", 3); // Animal 类型
```

### 1.14 ThisParameterType\<T\> 和 OmitThisParameter\<T\>

```typescript
// ThisParameterType：提取函数的 this 参数类型
type MyThisParameterType<T> = 
  T extends (this: infer U, ...args: never) => any ? U : unknown;

// OmitThisParameter：移除函数的 this 参数
type MyOmitThisParameter<T> = 
  unknown extends ThisParameterType<T> 
    ? T 
    : T extends (...args: infer A) => infer R 
      ? (...args: A) => R 
      : T;

// 使用场景
function toHex(this: Number) {
  return this.toString(16);
}

type T6 = ThisParameterType<typeof toHex>; // Number
type T7 = OmitThisParameter<typeof toHex>; // () => string
```

## 2. 类型递归

### DeepPartial 和 DeepReadonly

```typescript
// 递归地将所有属性变为可选
type DeepPartial<T> = T extends object
  ? { [K in keyof T]?: DeepPartial<T[K]> }
  : T;

// 递归地将所有属性变为只读
type DeepReadonly<T> = T extends object
  ? { readonly [K in keyof T]: DeepReadonly<T[K]> }
  : T;

// 扁平化嵌套数组类型
type Flatten<T> = T extends Array<infer U> ? Flatten<U> : T;
type F = Flatten<number[][][]>; // number
```

### 递归深度限制

TypeScript 对递归类型有深度限制（约 50 层）。超过时需要用尾递归优化：

```typescript
// 用辅助元组累积结果实现尾递归
type TupleOf<T, N extends number, R extends T[] = []> =
  R["length"] extends N ? R : TupleOf<T, N, [...R, T]>;

type FiveStrings = TupleOf<string, 5>; // [string, string, string, string, string]
```

## 3. 元组操作

```typescript
// Push：向元组末尾添加元素
type Push<T extends any[], V> = [...T, V];
type P = Push<[1, 2], 3>; // [1, 2, 3]

// Pop：移除末尾元素
type Pop<T extends any[]> = T extends [...infer Rest, any] ? Rest : never;

// Shift：移除首个元素
type Shift<T extends any[]> = T extends [any, ...infer Rest] ? Rest : never;

// Unshift：在开头添加元素
type Unshift<T extends any[], V> = [V, ...T];

// 元组转联合类型
type TupleToUnion<T extends any[]> = T[number];
type U = TupleToUnion<["a", "b", "c"]>; // "a" | "b" | "c"
```

## 4. 字符串类型操作

```typescript
// 字符串分割
type Split<S extends string, D extends string> =
  S extends `${infer Head}${D}${infer Tail}`
    ? [Head, ...Split<Tail, D>]
    : [S];

type Parts = Split<"a.b.c", ".">; // ["a", "b", "c"]

// 驼峰转换
type CamelCase<S extends string> =
  S extends `${infer Head}_${infer Tail}`
    ? `${Head}${Capitalize<CamelCase<Tail>>}`
    : S;

type Camel = CamelCase<"hello_world_foo">; // "helloWorldFoo"
```

## 5. 实战类型挑战

### 类型安全的 EventEmitter

```typescript
type EventMap = Record<string, any>;

interface TypedEventEmitter<Events extends EventMap> {
  on<K extends keyof Events>(event: K, listener: (data: Events[K]) => void): this;
  emit<K extends keyof Events>(event: K, data: Events[K]): boolean;
  off<K extends keyof Events>(event: K, listener: (data: Events[K]) => void): this;
}

// 使用
interface AppEvents {
  "user:login": { userId: string; timestamp: number };
  "user:logout": { userId: string };
  "error": { code: number; message: string };
}

declare const emitter: TypedEventEmitter<AppEvents>;
emitter.on("user:login", (data) => {
  // data 自动推断为 { userId: string; timestamp: number }
  console.log(data.userId);
});
```

## 6. 类型体操的实际价值

**什么时候该用：**
- 库/框架开发中提供精确的类型推断
- 消除重复的类型定义
- 确保类型约束在编译时被检查

**什么时候不该用：**
- 业务代码中过度复杂的类型（可读性优先）
- 能用简单联合类型解决的场景
- 团队成员无法理解和维护的类型
